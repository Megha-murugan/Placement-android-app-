│ 📊 DATABASE QUERIES (If Asked)                                  │
└─────────────────────────────────────────────────────────────────┘

"Show me all students":
  SELECT * FROM students;

"Show me all jobs":
  SELECT * FROM jobs;

"Show me applications":
  SELECT s.s_name, j.title, r.applied_at 
  FROM result r
  JOIN students s ON r.s_id = s.s_id
  JOIN jobs j ON r.j_id = j.j_id;

"How many applications per job":
  SELECT j.title, COUNT(r.r_id) as applications
  FROM jobs j
  LEFT JOIN result r ON j.j_id = r.j_id
  GROUP BY j.j_id;
